721  }
722  else

729  case 'EX:NIL':

731  {

734  $GLOBALS['_xh']['lv']=3;

738  case 'PARAMS':
739  case 'FAULT':
740  case 'METHODCALL':
741  case 'METHORESPONSE':

744  // End of INVALID ELEMENT!
745  // shall we add an assert here for unreachable code???
