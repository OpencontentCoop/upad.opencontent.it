662  }
663  else
664  {
665  $headers[$header_name] = trim( $arr[1] );
666  }
667  }
668  }
669  elseif( isset( $header_name ) )
670  {
671  /// @todo we should test that 1st char of line is either a space or tab
672  /// ('folding of header lines' in rfc 1945)
673  /// and possibly collapse leading whitspace
674  $headers[$header_name] .= ' ' . trim( $line );
675  }
676  }
677 
678  $data = substr( $data, $bd );
679 
680  // if CURL was used for the call, http headers have been processed,
681  // and dechunking + reinflating have been carried out
682  if( !$headers_processed )
683  {
684  // Decode chunked encoding sent by http 1.1 servers
685  if( isset( $headers['transfer-encoding'] ) && $headers['transfer-encoding'] == 'chunked' )
686  {
687  if( !$data = self::decode_chunked( $data ) )
688  {
689  $this->errorNumber = self::ERROR_BAD_RESPONSE;
690  $this->errorString = "Errors occurred when trying to rebuild the chunked data received from server.";
691  return false;
692  }
693  }
694 
695  // Decode gzip-compressed stuff
696  // code shamelessly inspired from nusoap library by Dietrich Ayala
697  if( isset( $headers['content-encoding'] ) )
698  {
699  $headers['content-encoding'] = str_replace( 'x-', '', $headers['content-encoding'] );
700  if( $headers['content-encoding'] == 'deflate' || $headers['content-encoding'] == 'gzip' )
701  {
702  // if decoding works, use it. else assume data wasn't gzencoded
703  if( function_exists( 'gzinflate' ) )
704  {
705  // RFC 2616 defines 'deflate' encoding as zlib format from RFC 1950,
706  // while many applications send raw deflate stream from RFC 1951.
707  // We check for presence of zlib header and use gzuncompress() or
708  // gzinflate() as needed. Code taken from Pear http2
709  if( $headers['content-encoding'] == 'deflate' )
710  {
711  $header = unpack( 'n', substr( $data, 0, 2 ) );
712  $degzdata = ( ( 0 == $header[1] % 31 ) ? @gzuncompress( $data ) : @gzinflate( $data ) );
713  if ( $degzdata )
714  {
715  $data = $degzdata;
716  }
717  }
718  elseif( $headers['content-encoding'] == 'gzip' && $degzdata = @gzinflate( substr( $data, 10 ) ) )
719  {
720  $data = $degzdata;
721  }
722  else
723  {
724  $this->errorNumber = self::ERROR_BAD_RESPONSE;
725  $this->errorString = "Errors occurred when trying to decode the deflated data received from server.";
726  return false;
727  }
728  }
729  else
730  {
731  $this->errorNumber = self::ERROR_BAD_RESPONSE;
732  $this->errorString = "The server sent deflated data. This php install must have the Zlib extension compiled in to support this.";
733  return false;
734  }
735  }
736  }
737  } // end of 'if needed, de-chunk, re-inflate response'
738 
739  return array( 'headers' => $headers, 'cookies' => $cookies, 'status_code' => $status_code );
740  }

742  /**
743  * Decode a string that is encoded w. "chunked" transfer encoding
744  * as defined in rfc2068 par. 19.4.6
745  * Code shamelessly stolen from nusoap library by Dietrich Ayala
746  *
747  * @param string $buffer the string to be decoded
748  * @return string
749  */
750  protected static function decode_chunked( $buffer )
751  {
752  $length = 0;
753  $new = '';
754 
755  // read chunk-size, chunk-extension (if any) and crlf
756  // get the position of the linebreak
757  $chunkend = strpos($buffer,"\r\n") + 2;
758  $temp = substr($buffer,0,$chunkend);
759  $chunk_size = hexdec( trim($temp) );
760  $chunkstart = $chunkend;
761  while($chunk_size > 0)
762  {
763  $chunkend = strpos($buffer, "\r\n", $chunkstart + $chunk_size);
764 
765  // just in case we got a broken connection
766  if($chunkend == false)
767  {
768  $chunk = substr($buffer,$chunkstart);
769  // append chunk-data to entity-body
770  $new .= $chunk;
771  $length += strlen($chunk);
772  break;
773  }
774 
775  // read chunk-data and crlf
776  $chunk = substr($buffer,$chunkstart,$chunkend-$chunkstart);
777  // append chunk-data to entity-body
778  $new .= $chunk;
779  // length := length + chunk-size
780  $length += strlen($chunk);
781  // read chunk-size and crlf
782  $chunkstart = $chunkend + 2;
783 
784  $chunkend = strpos($buffer,"\r\n",$chunkstart)+2;
785  if($chunkend == false)
786  {
787  break; //just in case we got a broken connection
788  }
789  $temp = substr($buffer,$chunkstart,$chunkend-$chunkstart);
790  $chunk_size = hexdec( trim($temp) );
791  $chunkstart = $chunkend;
792  }
793  return $new;
794  }
