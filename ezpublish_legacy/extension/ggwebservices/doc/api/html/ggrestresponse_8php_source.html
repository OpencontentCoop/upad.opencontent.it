28  }
29  if ( $contentType == '' )

32  }
33 
34  if ( $this->IsFault )
35  {
36  // default representation of an error, hand picked
37  $value = array( 'faultCode' => $this->FaultCode, 'faultString' => $this->FaultString );
38  // send an HTTP error code, since there is no other way to make sure
39  // that the client can tell apart error responses from valid array responses
40  // try some meaningful mapping (it's REST, baby!)
41  switch( $this->FaultCode )
42  {
43  case ggWebservicesServer::INVALIDMETHODERROR:
44  header( $_SERVER['SERVER_PROTOCOL'].' 404 Not Found' );
45  break;
46  case ggWebservicesServer::INVALIDPARAMSERROR:
47  case ggWebservicesServer::INVALIDCOMPRESSIONERROR:
48  case ggWebservicesServer::INVALIDREQUESTERROR:
49  header( $_SERVER['SERVER_PROTOCOL'].' 400 Bad Request' );

52  header( $_SERVER['SERVER_PROTOCOL'].' 403 Forbidden' );
53  break;
54  default:
55  header( $_SERVER['SERVER_PROTOCOL'].' 500 Internal Server Error' );
56  }
57  }
58  else

60  $value = $this->Value;
61  }
62 
63  switch( $contentType )
64  {
65  case 'php':
66  case 'application/x-httpd-php':
67  return var_export( $value );
68  case 'phps':
69  case 'application/vnd.php.serialized':
70  return serialize( $value );
71  case 'application/json':

73  if ( $this->JsonpCallback != false )
74  {
75  $json = $this->JsonpCallback . '(' . $json . ')';
76  }
77  return $json;
78  case 'text/plain':

80  case 'text/html':

83  default:
